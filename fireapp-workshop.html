p.65 Partial

<html>
<head>
  <%= render :partial => "head" %>
</head>
<body>
  <%= render :partial => "header" %>
  <%= render :partial => "main_menu" %>
  <div class="main-block">
    <div class="main">
      <%= yield %>
    </div>
    <div class="side">
      <div class="ad">
        側欄宣傳文字
      </div>
      <div class="links">
        <h2>相關資訊連結</h2>
        <ul>
          <% 8.times do %>

          <% end %>

      </div>
    </div>
  </div>

</html>
